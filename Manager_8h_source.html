21struct Node {

28 Node(BDD_ID id, std::string label) : id(id), label(label){};
29
30 bool isConstant() { return id == high && id == low && id == top; }
31 bool isVariable() { return !isConstant() && top == id; }
32
33 bool operator==(const Node& rhs) const {
34 return high == rhs.high && low == rhs.low && top == rhs.top;
35 }

48struct TupleHasher {
49 std::size_t operator()(const Key& key) const {
50 std::size_t seed = 0;
51
52 // std::get<2>(key) << 21 | std::get<1>(key) << 42 | std::get<0>(key);
53 boost::hash_combine(seed, boost::hash_value(std::get<0>(key)));
54 boost::hash_combine(seed, boost::hash_value(std::get<1>(key)));
55 boost::hash_combine(seed, boost::hash_value(std::get<2>(key)));
56
57 return seed;
58 }
59};
