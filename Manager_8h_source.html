21struct Node {
22 BDD_ID id;
23 BDD_ID high;
24 BDD_ID low;
25 BDD_ID top;
26 std::string label;
27
28 Node(BDD_ID id, std::string label) : id(id), label(label){};
29
30 bool isConstant() { return id == high && id == low && id == top; }
31 bool isVariable() { return !isConstant() && top == id; }
32
33 bool operator==(const Node& rhs) const {
34 return high == rhs.high && low == rhs.low && top == rhs.top;
35 }
36};

38struct TupleHasher {
39 std::size_t operator()(const Key& key) const {
40 std::size_t seed = 0;
41
42 // std::get<2>(key) << 21 | std::get<1>(key) << 42 | std::get<0>(key);
43 boost::hash_combine(seed, boost::hash_value(std::get<0>(key)));
44 boost::hash_combine(seed, boost::hash_value(std::get<1>(key)));
45 boost::hash_combine(seed, boost::hash_value(std::get<2>(key)));
46
47 return seed;
48 }
49};
