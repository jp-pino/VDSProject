19 Node(BDD_ID id, std::string label) : id(id), label(label){};

21 bool isConstant() { return id == high && id == low && id == top; }
22 bool isVariable() { return !isConstant(); }
23
24 bool operator==(const Node& rhs) const {
25 return high == rhs.high && low == rhs.low && top == rhs.top;
